    Word Break Problem
Given an input string and a dictionary of words, find out if the input string can be segmented
into a space-separated sequence of dictionary words. See following examples for more details.
This is a famous Google interview question, also being asked by many other companies now a days.
Consider the following dictionary
	{ i, like, sam, sung, samsung, mobile, ice, cream, icecream, man, go, mango}
	Input:  ilike
	Output: Yes
	The string can be segmented as "i like".

	Input:  ilikesamsung
	Output: Yes
	The string can be segmented as "i like samsung" or "i like sam sung".

class BasePage{

    page

    scrollToTop

    scrollToButtom

    getTitle

    getUrl
    ...
}

class LoginPage extends basepage{

    constrcutor(){
        this.usernameInputBox = locator("#username");
        this.passwordInputBox = locator("#password");
        this.loginButton = locator("button[name='login']);
    }

    public async enterUsername(username){
        this.usernameInputBox.fill(username);
    }

    public async enterPassword(password){
        this.passwordInputBox.fill(password);
    }

    public async clickLoginButton(){
        this.loginButton.click();
    }

    public async login(username, password) {
        this.enterUsername(username);
        this.enterPassword(password);
        this.click();
    }

}



test(){ // with POM
    LoginPage loginPage = new LoginPage();

    loginPage.login();
}
